        function startGame() {
            if (!selectedDifficulty || !currentWeaponKey || !selectedTeamSize) return;

        function initThree() {
            const container = document.getElementById('canvas-container');
            scene = new THREE.Scene();
            scene.background = new THREE.Color(0xdce9f5);

            camera = new THREE.PerspectiveCamera(60, window.innerWidth / window.innerHeight, 0.05, 1000);
            camera.rotation.order = 'YXZ';
            renderer = new THREE.WebGLRenderer({ antialias: true });
            renderer.setSize(window.innerWidth, window.innerHeight);
            container.appendChild(renderer.domElement);

            raycaster = new THREE.Raycaster();

            const ambientLight = new THREE.AmbientLight(0xffffff, 0.85);
            scene.add(ambientLight);
            const dirLight = new THREE.DirectionalLight(0xffffff, 0.6); dirLight.position.set(10, 40, 15); scene.add(dirLight);
            const dirLight2 = new THREE.DirectionalLight(0xaaccff, 0.4); dirLight2.position.set(-10, 30, -15);
            scene.add(dirLight2);

            const floorGeo = new THREE.PlaneGeometry(MAP_LIMIT * 2, MAP_LIMIT * 2);
            floorGeo.rotateX(-Math.PI / 2);
            const floorMat = new THREE.MeshLambertMaterial({ color: 0xeef2f7 });
            const floor = new THREE.Mesh(floorGeo, floorMat);
            scene.add(floor);

            const wallMat = new THREE.MeshLambertMaterial({ color: 0xd1dbe5 });
            scene.add(createWall(MAP_LIMIT*2, 8, 1, 0, 4, MAP_LIMIT, wallMat));
            scene.add(createWall(MAP_LIMIT*2, 8, 1, 0, 4, -MAP_LIMIT, wallMat));
            scene.add(createWall(1, 8, MAP_LIMIT*2, MAP_LIMIT, 4, 0, wallMat));

            initChainsawMesh();

            window.addEventListener('keydown', e => handleKey(e, true));
            window.addEventListener('keyup', e => handleKey(e, false));

            window.addEventListener('mouseup', () => { keys.e = false; });

        function setupRound() {
            roundProcessing = false;
            document.getElementById('round-alert').style.display = 'none';

            player.y = 0;
            player.isJumping = false;
            player.isSliding = false;

            player.rotationY = 0;

        function createWall(w, h, d, x, y, z, mat) {
            const mesh = new THREE.Mesh(new THREE.BoxGeometry(w, h, d), mat);
            mesh.position.set(x, y, z);
            obstacleMeshes.push(mesh);
            obstacles.push({ x: x, z: z, radius: (Math.max(w, d) / 2) + 0.4, y: y, h: h, type: 'box', w: w, d: d });
            return mesh;
        }

        function generateArenaStructures() {
            const mat = new THREE.MeshLambertMaterial({ color: 0xbacad6 });
            const cylinderMat = new THREE.MeshLambertMaterial({ color: 0xa4b5c4 });

            createObstacleBox(8, 4, 8, 0, 2, 0, mat);
            createObstacleBox(12, 2, 2, 0, 1, -5, mat);
            createObstacleBox(12, 2, 2, 0, 1, 5, mat);
            createObstacleBox(6, 3, 2, -12, 1.5, -12, mat);
            createObstacleBox(2, 3, 6, -14, 1.5, -8, mat);
            createObstacleBox(6, 3, 2, 12, 1.5, 12, mat);
            createObstacleBox(2, 3, 6, 14, 1.5, 8, mat);
            createObstacleBox(6, 3, 2, 12, 1.5, -12, mat);
            createObstacleBox(2, 3, 6, 14, 1.5, -8, mat);
            createObstacleBox(6, 3, 2, -12, 1.5, 12, mat);
            createObstacleBox(2, 3, 6, -14, 1.5, 8, mat);
            createObstacleBox(3, 4, 8, -26, 2, 0, mat);
            createObstacleBox(3, 4, 8, 26, 2, 0, mat);

            createCylinderObstacle(1.8, 4, -22, 2, -18, cylinderMat);
            createCylinderObstacle(1.8, 4, 22, 2, 18, cylinderMat);
            createCylinderObstacle(1.8, 4, -22, 2, 18, cylinderMat);
            createCylinderObstacle(1.8, 4, 22, 2, -18, cylinderMat);
            createCylinderObstacle(2.5, 3, -6, 1.5, -20, cylinderMat);
            createCylinderObstacle(2.5, 3, 6, 1.5, 20, cylinderMat);

            createObstacleBox(6, 2, 4, 24, 1, -8, mat);
            createSlope(4, 2, 6, 24, 1, -3, Math.PI, mat);

            createObstacleBox(6, 2, 4, -24, 1, 8, mat);
            createSlope(4, 2, 6, -24, 1, 3, 0, mat);
        }

        function createObstacleBox(w, h, d, x, y, z, mat) {
            const mesh = new THREE.Mesh(new THREE.BoxGeometry(w, h, d), mat);
            mesh.position.set(x, y, z);
            scene.add(mesh);
            obstacleMeshes.push(mesh);
            obstacles.push({ x: x, z: z, y: y, h: h, type: 'box', w: w, d: d });
        }

        function createCylinderObstacle(radius, h, x, y, z, mat) {
            const mesh = new THREE.Mesh(new THREE.CylinderGeometry(radius, radius, h, 12), mat);
            mesh.position.set(x, y, z);
            scene.add(mesh);
            obstacleMeshes.push(mesh);
            obstacles.push({ x: x, z: z, radius: radius, y: y, h: h, type: 'cylinder' });
        }

        function createSlope(w, h, d, x, y, z, rotY, mat) {
            const group = new THREE.Group();
            const slopeGeo = new THREE.BoxGeometry(w, 0.2, Math.sqrt(h*h + d*d));
            const slopeMesh = new THREE.Mesh(slopeGeo, mat);
            slopeMesh.position.set(0, 0, 0);
            slopeMesh.rotation.x = -Math.atan2(h, d);
            group.add(slopeMesh);
            group.position.set(x, y, z);
            group.rotation.y = rotY;
            scene.add(group);

            obstacleMeshes.push(slopeMesh);
            obstacles.push({ x: x, z: z, y: y, h: h, type: 'slope', w: w, d: d, rotY: rotY });
        }

        function initPlayerHitboxes() {
            const hitMat = new THREE.MeshBasicMaterial({ visible: false });
            player.headMesh = new THREE.Mesh(new THREE.BoxGeometry(0.4, 0.4, 0.4), hitMat);
            player.headMesh.userData = { isPlayer: true, isHead: true };
            scene.add(player.headMesh);
            player.bodyMesh = new THREE.Mesh(new THREE.BoxGeometry(0.8, 1.2, 0.6), hitMat);
            player.bodyMesh.userData = { isPlayer: true, isBody: true };
            scene.add(player.bodyMesh);
        }

        function updatePlayerHitboxes() {
            if (!player.headMesh || !player.bodyMesh) return;
            player.headMesh.position.copy(camera.position);
            player.bodyMesh.position.copy(camera.position);
            
            if (player.isSliding) {

        function handleKey(e, isDown) {
            if (!gameActive || player.hp <= 0 || roundProcessing) return;
            let key = e.key.toLowerCase();
            if ((e.key === ' ' || e.code === 'Space') && isDown && !player.isJumping && !player.isSliding) {
                player.jumpVelocity = 0.43;
                player.isJumping = true;
                return;
            }
            if (key === 'r' && isDown && !player.isSliding && currentWeaponKey !== 'chainsaw') {
                startReload();
                return;
            }
            if (key === 'c' && isDown && !player.isSliding && !player.isJumping && player.slideCooldown <= 0) {
                let moveVector = new THREE.Vector3();
                if (keys.w) moveVector.z -= 1; if (keys.s) moveVector.z += 1;
                if (keys.a) moveVector.x -= 1; if (keys.d) moveVector.x += 1;
                if (moveVector.lengthSq() === 0) moveVector.z = -1;
                moveVector.normalize().applyQuaternion(camera.quaternion);
                moveVector.y = 0; moveVector.normalize();

                player.isSliding = true;
                player.slideTimer = player.slideDuration;
                player.slideDir.copy(moveVector);
                return;
            }
            if (keys.hasOwnProperty(key)) keys[key] = isDown;
        }

        function checkAndFirePlayerWeapon() {
            if (isReloading || player.isSliding || player.hp <= 0 || !gameActive || roundProcessing) return;

            const targetDirection = new THREE.Vector3(0, 0, -1).applyQuaternion(camera.quaternion).normalize();
            raycaster.set(camera.position, targetDirection);
            
            let targets = [];
            enemies.forEach(e => { if(e.hp > 0) targets.push(e.headMesh, e.targetMesh); });
            let targetsAndWalls = targets.concat(obstacleMeshes);

            const intersects = raycaster.intersectObjects(targetsAndWalls);

            if (intersects.length > 0) {
                const hitObj = intersects[0].object;
                const w = WEAPONS[currentWeaponKey];

                if (currentWeaponKey !== 'chainsaw' && hitObj.userData.isAI && hitObj.userData.botRef.team === 'enemy' && intersects[0].distance <= w.range) {
                    executePlayerShot(targetDirection, intersects);
                } else if (keys.e) {
                    executePlayerShot(targetDirection, intersects);
                }
            } else if (keys.e) {
                executePlayerShot(targetDirection, []);
            }
        }

        function executePlayerShot(targetDirection, intersects) {
            const now = performance.now();
            const w = WEAPONS[currentWeaponKey];
            if (now - lastFireTime < w.fireRate) return;
            lastFireTime = now;

            if (currentWeaponKey !== 'chainsaw') {
                if (w.currentMag <= 0) { startReload(); return; }
                w.currentMag--;
            } else {
                w.currentMag = Math.min(150, w.currentMag + 1);
            }
            updateWeaponUI();

            if (intersects.length > 0 && intersects[0].distance <= w.range) {
                const hitObj = intersects[0].object;

                if (obstacleMeshes.includes(hitObj)) {
                    if (currentWeaponKey !== 'chainsaw') {
                        createLineEffect(camera.position, targetDirection, intersects[0].distance, 0xffea00, 3);
                    }
                    createHitParticle(intersects[0].point, new THREE.Color(0xbacad6));
                    return;
                }

                let bot = hitObj.userData.botRef;
                if (bot && bot.team === 'enemy' && bot.hp > 0) {
                    let isHead = hitObj.userData.isHead;
                    let dmg = Math.floor(Math.random() * (w.damageMax - w.damageMin + 1)) + w.damageMin;
                    if (isHead && currentWeaponKey !== 'chainsaw') dmg = Math.floor(dmg * 1.7);

                    bot.hp -= dmg;
                    
                    if (currentWeaponKey !== 'chainsaw') {
                        let color = isHead ? 0xff3300 : 0xffea00;
                        createLineEffect(camera.position, targetDirection, intersects[0].distance, color, isHead ? 5 : 3);
                    }
                    createHitParticle(intersects[0].point, new THREE.Color(0xff3300));
                    popDamageText(dmg, intersects[0].point, isHead);

                    if (bot.hp <= 0) {
                        bot.hp = 0;
                        scene.remove(bot.mesh);
                        updateTeamUI();
                        checkRoundEnd();
                    }
                }
            } else {
                if (currentWeaponKey !== 'chainsaw') {
                    createLineEffect(camera.position, targetDirection, w.range, 0xaaaaaa, 1);
                }
            }
        }

        function checkAndFireChainsawSpecial(now) {
            if (currentWeaponKey !== 'chainsaw' || !keys.q || player.hp <= 0 || !gameActive || roundProcessing) return;

            const w = WEAPONS.chainsaw;
            if (w.currentMag <= 0) {
                w.currentMag = 0;
                updateWeaponUI();
                return;
            }

            if (now - lastChainsawQTime >= 40) {
                lastChainsawQTime = now;
                w.currentMag = Math.max(0, w.currentMag - 1);
                updateWeaponUI();

                const targetDirection = new THREE.Vector3(0, 0, -1).applyQuaternion(camera.quaternion).normalize();
                raycaster.set(camera.position, targetDirection);

                let targetsAndWalls = targets.concat(obstacleMeshes);
                const intersects = raycaster.intersectObjects(targetsAndWalls);

                if (intersects.length > 0 && intersects[0].distance <= w.range) {
                    const hitObj = intersects[0].object;

                    if (obstacleMeshes.includes(hitObj)) {
                        createHitParticle(intersects[0].point, new THREE.Color(0xffaa00));

                            checkRoundEnd();

            }
        }

        function createLineEffect(startPos, dir, dist, hexColor, width) {
            const endPos = startPos.clone().add(dir.clone().multiplyScalar(dist));
            const geo = new THREE.BufferGeometry().setFromPoints([startPos.clone().sub(new THREE.Vector3(0,0.1,0)), endPos]);
            const mat = new THREE.LineBasicMaterial({ color: hexColor, linewidth: width, transparent: true, opacity: 0.8 });
            const line = new THREE.Line(geo, mat);
            scene.add(line);
            particles.push({ mesh: line, vel: new THREE.Vector3(), life: 4 });
        }

        function createHitParticle(point, color) {
            for (let i = 0; i < 6; i++) {
                const pGeo = new THREE.BoxGeometry(0.06, 0.06, 0.06);
                const pMat = new THREE.MeshBasicMaterial({ color: color, transparent: true, opacity: 0.9 });
                const mesh = new THREE.Mesh(pGeo, pMat);
                mesh.position.copy(point);
                scene.add(mesh);

                let vel = new THREE.Vector3((Math.random()-0.5)*0.25, (Math.random())*0.28, (Math.random()-0.5)*0.25);
                particles.push({ mesh: mesh, vel: vel, life: 15 + Math.floor(Math.random()*15) });
            }
        }

        function popDamageText(damage, worldPos, isHead) {
            const proj = worldPos.clone().project(camera);
            if (proj.z > 1) return;

        }

        function updateAllAI(now) {
            let activeEnemies = enemies.filter(e => e.hp > 0);
            let activeAllies = allies.filter(a => a.hp > 0);
            let playerAlive = player.hp > 0;

            activeEnemies.forEach(bot => updateSingleAI(bot, activeAllies, playerAlive, now));
            activeAllies.forEach(bot => updateSingleAI(bot, activeEnemies, false, now));
        }

        function updateSingleAI(bot, opposingList, playerAlive, now) {
            let possibleTargets = [];
            opposingList.forEach(t => possibleTargets.push({ pos: t.mesh.position, ref: t, isPlayer: false }));
            if (playerAlive) possibleTargets.push({ pos: camera.position, ref: player, isPlayer: true });

            if (possibleTargets.length === 0) return;

            let closest = null; let minDist = 9999;
            possibleTargets.forEach(t => {
                let d = bot.mesh.position.distanceTo(t.pos);
                if (d < minDist) { minDist = d; closest = t; }
            });

            if (!closest) return;

            let targetPos = closest.pos.clone();
            let dist = bot.mesh.position.distanceTo(targetPos);

            if (dist > 3.0) {
                let dir = targetPos.clone().sub(bot.mesh.position); dir.y = 0; dir.normalize();
                bot.moveDir.copy(dir);
            } else {
                bot.moveDir.set(0,0,0);
            }

            bot.strafeTimer -= 16.66;
            if (bot.strafeTimer <= 0) {
                bot.strafeTimer = 600 + Math.random()*1000;
                bot.strafeDir = Math.random() > 0.5 ? 1 : -1;
            }
            if (bot.moveDir.lengthSq() > 0) {
                let side = new THREE.Vector3(-bot.moveDir.z, 0, bot.moveDir.x).normalize();
                bot.moveDir.add(side.multiplyScalar(bot.strafeDir * 0.4)).normalize();
            }

            let nextX = bot.mesh.position.x + bot.moveDir.x * bot.speed;
            let nextZ = bot.mesh.position.z + bot.moveDir.z * bot.speed;
            if (!checkCollisionAt(nextX, nextZ, 0.45)) {
                bot.mesh.position.x = Math.max(-MAP_LIMIT+1, Math.min(MAP_LIMIT-1, nextX));
                bot.mesh.position.z = Math.max(-MAP_LIMIT+1, Math.min(MAP_LIMIT-1, nextZ));
            }

            // AIの高さ（乗る処理）
            let standY = getFloorHeightAt(bot.mesh.position.x, bot.mesh.position.z);
            bot.mesh.position.y = standY;

            bot.mesh.lookAt(new THREE.Vector3(targetPos.x, bot.mesh.position.y, targetPos.z));

            if (dist < 45) {
                bot.shootCooldown -= 16.66;
                if (bot.shootCooldown <= 0) {
                    let fireWep = WEAPONS[bot.weapon];
                    bot.shootCooldown = fireWep.fireRate + Math.random()*120;

                    let aimAt = targetPos.clone();
                    if (closest.isPlayer) aimAt.y = camera.position.y - 0.2;
                    else aimAt.y = targetPos.y + 0.9;

                    aimAt.x += (Math.random()-0.5)*bot.aimError * dist;
                    aimAt.z += (Math.random()-0.5)*bot.aimError * dist;

                    let shootDir = aimAt.clone().sub(bot.mesh.position.clone().add(new THREE.Vector3(0, bot.eyeHeight, 0))).normalize();
                    executeAIShot(bot, shootDir, dist, closest);
                }
            }
        }

        function executeAIShot(bot, dir, maxDist, targetObj) {
            let aiWep = WEAPONS[bot.weapon];
            let startPos = bot.mesh.position.clone().add(new THREE.Vector3(0, bot.eyeHeight, 0));
            let ray = new THREE.Raycaster(startPos, dir);
            
            let targetMeshes = [];
            if (targetObj.isPlayer) {
                targetMeshes.push(player.headMesh, player.bodyMesh);
            } else {
                targetMeshes.push(targetObj.ref.headMesh, targetObj.ref.targetMesh);
            }
            let alloc = targetMeshes.concat(obstacleMeshes);
            let hits = ray.intersectObjects(alloc);

            if (hits.length > 0 && hits[0].distance <= aiWep.range) {
                let first = hits[0].object;
                if (obstacleMeshes.includes(first)) {
                    createLineEffect(startPos, dir, hits[0].distance, 0xffffff, 1.5);
                    return;
                }
                
                createLineEffect(startPos, dir, hits[0].distance, 0xff5555, 2);

                if (targetObj.isPlayer) {
                    let dmg = Math.floor(Math.random() * (aiWep.damageMax - aiWep.damageMin + 1)) + aiWep.damageMin;
                    if (first.userData.isHead) dmg = Math.floor(dmg * 1.5);
                    
                    player.hp -= dmg;
                    triggerDamageIndicator(bot.mesh.position);

                    document.getElementById('hp-text').innerText = `PLAYER HP: ${Math.max(0, player.hp)} / ${player.maxHp}`;
                    document.getElementById('hp-bar').style.width = `${Math.max(0, (player.hp / player.maxHp)*100)}%`;

                    if (player.hp <= 0) {
                        player.hp = 0;
                        updateTeamUI();
                        checkRoundEnd();
                    }
                } else {
                    let allyBot = targetObj.ref;
                    let dmg = Math.floor(Math.random() * (aiWep.damageMax - aiWep.damageMin + 1)) + aiWep.damageMin;
                    allyBot.hp -= dmg;
                    if (allyBot.hp <= 0) {
                        allyBot.hp = 0;
                        scene.remove(allyBot.mesh);
                        updateTeamUI();
                        checkRoundEnd();
                    }
                }
            }
        }

        function triggerDamageIndicator(sourcePos) {

        // 高度および衝突判定の共通ヘルパー
        function getFloorHeightAt(x, z) {
            let highestY = 0;
            for (let i = 0; i < obstacles.length; i++) {
                let o = obstacles[i];
                if (o.type === 'box') {
                    let halfW = o.w / 2; let halfD = o.d / 2;
                    if (x >= o.x - halfW && x <= o.x + halfW && z >= o.z - halfD && z <= o.z + halfD) {
                        let topY = o.y + (o.h / 2);
                        if (topY > highestY) highestY = topY;
                    }
                } else if (o.type === 'cylinder') {
                    let dx = x - o.x; let dz = z - o.z;
                    if ((dx*dx + dz*dz) <= o.radius*o.radius) {
                        let topY = o.y + (o.h / 2);
                        if (topY > highestY) highestY = topY;
                    }
                } else if (o.type === 'slope') {
                    // 簡易的な坂のZ/X連動高度判定
                    let halfW = o.w / 2; let halfD = o.d / 2;
                    let localX = x - o.x; let localZ = z - o.z;
                    // 回転を考慮
                    if (o.rotY === Math.PI) { localZ = -localZ; localX = -localX; }
                    if (localX >= -halfW && localX <= halfW && localZ >= -halfD && localZ <= halfD) {
                        // 手前から奥にかけて高くなる計算
                        let ratio = (localZ + halfD) / o.d; 
                        let currentSlopeY = o.y - (o.h/2) + (ratio * o.h);
                        if (currentSlopeY > highestY) highestY = currentSlopeY;
                    }
                }
            }
            return highestY;
        }

        function checkCollisionAt(nx, nz, pad = 0.45) {
            for (let i = 0; i < obstacles.length; i++) {
                let o = obstacles[i];
                if (o.type === 'box') {
                    let halfW = o.w / 2 + pad; let halfD = o.d / 2 + pad;
                    if (nx > o.x - halfW && nx < o.x + halfW && nz > o.z - halfD && nz < o.z + halfD) {
                        // 現在のプレイヤー/AIのY座標が、オブジェクトの高さより低ければ衝突
                        let currentY = (pad === 0.45) ? camera.position.y - player.currentEyeHeight : 0; 
                        if (currentY < o.y + o.h/2 - 0.1) return true;
                    }
                } else if (o.type === 'cylinder') {
                    let dx = nx - o.x; let dz = nz - o.z;
                    let rSq = (o.radius + pad) * (o.radius + pad);
                    if ((dx*dx + dz*dz) < rSq) {
                        let currentY = (pad === 0.45) ? camera.position.y - player.currentEyeHeight : 0;
                        if (currentY < o.y + o.h/2 - 0.1) return true;
                    }
                }
            }
            return false;
        }

      function checkRoundEnd() {
            if (roundProcessing) return;
            let enemyAlive = enemies.some(e => e.hp > 0);
            let allyAlive = (player.hp > 0) || allies.some(a => a.hp > 0);

            if (!enemyAlive) {
                roundProcessing = true; scoreAlly++; updateScoreUI();
                processRoundEndSequence(true);
            } else if (!allyAlive) {
                roundProcessing = true; scoreEnemy++; updateScoreUI();
                processRoundEndSequence(false);
            }
        }

                if (scoreAlly >= 5) { endMatch(true); }
                else if (scoreEnemy >= 5) { endMatch(false); }

        }

        function animate() {
            if (!gameActive) return;
            requestAnimationFrame(animate);

            const now = performance.now();
            let delta = now - lastTimeUpdate;
            if (delta > 100) delta = 16.66;
            lastTimeUpdate = now;

            let lookSpeed = 0.032;
            if (keys.arrowleft) player.rotationY += lookSpeed;
            if (keys.arrowright) player.rotationY -= lookSpeed;
            if (keys.arrowup) player.rotationX = Math.max(-Math.PI/2.3, Math.min(Math.PI/2.3, player.rotationX + lookSpeed * 0.8));
            if (keys.arrowdown) player.rotationX = Math.max(-Math.PI/2.3, Math.min(Math.PI/2.3, player.rotationX - lookSpeed * 0.8));

            camera.quaternion.setFromEuler(new THREE.Euler(player.rotationX, player.rotationY, 0, 'YXZ'));

            let currentMoveSpeed = player.speed;
            if (currentWeaponKey === 'chainsaw' && keys.q && WEAPONS.chainsaw.currentMag > 0) {
                currentMoveSpeed = player.speed * 1.5; 
            }

            if (player.slideCooldown > 0) {
                player.slideCooldown -= delta;
                document.getElementById('slide-bar').style.width = `${Math.min(100, (1 - player.slideCooldown / player.slideCooldownMax) * 100)}%`;
                if (player.slideCooldown <= 0) document.getElementById('slide-text').innerText = "SLIDING READY";
            }

            // 現在の座標を基に床や壁の「高さ」を取得
            let floorTopY = getFloorHeightAt(camera.position.x, camera.position.z);

            if (player.isSliding) {
                player.slideTimer -= delta;
                player.currentEyeHeight = player.eyeHeight * 0.55;
                document.getElementById('slide-text').innerText = "SLIDING...";
                document.getElementById('slide-bar').style.width = '0%';

                let slSpeed = 0.38 * (player.slideTimer / player.slideDuration) + 0.06;
                let nextX = camera.position.x + player.slideDir.x * slSpeed;
                let nextZ = camera.position.z + player.slideDir.z * slSpeed;

                if (!checkCollisionAt(nextX, nextZ, 0.45)) {
                    camera.position.x = Math.max(-MAP_LIMIT+1, Math.min(MAP_LIMIT-1, nextX));
                    camera.position.z = Math.max(-MAP_LIMIT+1, Math.min(MAP_LIMIT-1, nextZ));
                } else {
                    player.isSliding = false;
                    player.slideCooldown = player.slideCooldownMax;
                }

                if (player.slideTimer <= 0) {
                    player.isSliding = false;
                    player.slideCooldown = player.slideCooldownMax;
                }
            } else {
                player.currentEyeHeight = player.currentEyeHeight * 0.8 + player.eyeHeight * 0.2;
                
                let moveX = 0, moveZ = 0;
                if (keys.w) moveZ -= 1; if (keys.s) moveZ += 1;
                if (keys.a) moveX -= 1; if (keys.d) moveX += 1;

                if (moveX !== 0 || moveZ !== 0) {
                    let moveVector = new THREE.Vector3(moveX, 0, moveZ).normalize();
                    moveVector.applyQuaternion(camera.quaternion);
                    moveVector.y = 0; moveVector.normalize();

                    let nextX = camera.position.x + moveVector.x * currentMoveSpeed;
                    let nextZ = camera.position.z + moveVector.z * currentMoveSpeed;

                    if (!checkCollisionAt(nextX, nextZ, 0.45)) {
                        camera.position.x = Math.max(-MAP_LIMIT+1, Math.min(MAP_LIMIT-1, nextX));
                        camera.position.z = Math.max(-MAP_LIMIT+1, Math.min(MAP_LIMIT-1, nextZ));
                    }
                }
            }

            // 重力とジャンプ・乗る処理
            if (player.isJumping) {
                player.jumpVelocity -= 0.021; 
                player.y += player.jumpVelocity;
                if (player.y <= floorTopY) { 
                    player.y = floorTopY; 
                    player.jumpVelocity = 0; 
                    player.isJumping = false; 
                }
            } else {
                // 落下処理、または坂道移動による自動追従
                if (player.y > floorTopY) {
                    player.y -= 0.15; // 簡易的な自由落下
                    if (player.y < floorTopY) player.y = floorTopY;
                } else {
                    player.y = floorTopY;
                }
            }
            camera.position.y = player.y + player.currentEyeHeight;

            if (currentWeaponKey === 'chainsaw' && chainsawMesh) {
                if (keys.q && WEAPONS.chainsaw.currentMag > 0) {
                    chainsawMesh.position.x = 0.32 + Math.sin(now * 0.15) * 0.015;
                    chainsawMesh.position.y = -0.28 + Math.cos(now * 0.15) * 0.015;
                    chainsawMesh.position.z = -0.48;
                    chainsawMesh.rotation.set(0.1, 0, 0);
                } else if (keys.e || (now - lastFireTime < WEAPONS.chainsaw.fireRate)) {
                    let wave = Math.sin((now - lastFireTime) * 0.02);
                    chainsawMesh.position.x = 0.15 + wave * 0.4;
                    chainsawMesh.position.y = -0.25 + Math.abs(wave) * 0.2;
                    chainsawMesh.rotation.set(0.2, wave * 0.8, wave * 0.5);
                } else {
                    chainsawMesh.position.x = 0.28 + Math.sin(now * 0.003) * 0.01;
                    chainsawMesh.position.y = -0.28 + Math.cos(now * 0.004) * 0.008;
                    chainsawMesh.position.z = -0.5;
                    chainsawMesh.rotation.set(0, 0, 0);
                }
            }

            if (isReloading) {
                let w = WEAPONS[currentWeaponKey];
                let elapsed = now - reloadStartTime;
                document.getElementById('reload-bar').style.width = `${Math.min(100, (elapsed / w.reloadTime) * 100)}`;
                if (elapsed >= w.reloadTime) {
                    isReloading = false;
                    document.getElementById('reload-container').style.display = 'none';
                    let fill = Math.min(w.maxMagazine - w.currentMag, w.currentReserve);
                    w.currentMag += fill; w.currentReserve -= fill;
                    updateWeaponUI();
                }
            }

            updatePlayerHitboxes();
            checkAndFirePlayerWeapon();          
            checkAndFireChainsawSpecial(now);    
            updateAllAI(now);

            for (let i = particles.length - 1; i >= 0; i--) {
                let p = particles[i]; p.mesh.position.add(p.vel); p.life--;
                if(p.life <= 0){ scene.remove(p.mesh); particles.splice(i, 1); }
            }

            renderer.render(scene, camera);
        }
    </script>
